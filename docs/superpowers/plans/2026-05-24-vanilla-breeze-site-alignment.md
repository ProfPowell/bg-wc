# gl-wc vanilla-breeze Site Alignment Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Bring gl-wc in line with the author's vanilla-breeze sibling-repo standard — npm-installed `vanilla-breeze` + `@profpowell/code-block`, a Vite build (library via `preserveModules`, site as a multi-page app), top-level `docs/` + `demos/`, cem analyze, Playwright + eslint + prettier, and a standard Pages deploy.

**Architecture:** Two Vite invocations share one repo. `vite.config.js` builds the **library** to `dist/` with `output.preserveModules: true`, so `dist/` mirrors `src/` and the 55 dynamic-`import()` preset chunks keep stable, unhashed names (preserving the `./presets/*` export contract). `vite.site.config.js` builds the **product site** (`docs/` + `demos/`) as a multi-page app, resolving bare imports from `node_modules`. The product site (gallery + API docs) is driven by vanilla-breeze tokens/themes and renders every snippet with `<code-block>`; the 32 demos stay bespoke. One small library change bridges vanilla-breeze's `--color-text` to gl-wc's `--color-foreground` read.

**Tech Stack:** Vite 7, vanilla-breeze, @profpowell/code-block, @custom-elements-manifest/analyzer, Playwright, ESLint 9, Prettier.

**Reference design:** `docs/superpowers/specs/2026-05-24-vanilla-breeze-site-alignment-design.md`

```js
export function readTokenString(host, token, override) {
  const cs = getComputedStyle(host);
  if (override) {
    const v = cs.getPropertyValue(override).trim();
    if (v) return v;
  }
  const tokens = Array.isArray(token) ? token : [token];
  for (const t of tokens) {
    const v = cs.getPropertyValue(t).trim();
    if (v) return v;
  }
  return DEFAULTS[tokens[0]] || '#000';
}
```

```js
function stripPrivateMembers() {
  return {
    name: 'strip-private-members',
    packageLinkPhase({ customElementsManifest }) {
      for (const mod of customElementsManifest.modules || []) {
        for (const decl of mod.declarations || []) {
          if (decl.members) {
            decl.members = decl.members.filter(
              (m) => !m.name.startsWith('_') && !m.name.startsWith('#')
            );
          }
        }
      }
    },
  };
}

function addCssParts() {
  return {
    name: 'add-css-parts',
    packageLinkPhase({ customElementsManifest }) {
      for (const mod of customElementsManifest.modules || []) {
        for (const decl of mod.declarations || []) {
          if (decl.tagName === 'gl-wc') {
            decl.cssParts = [
              { name: 'canvas', description: 'The rendered background canvas element' },
            ];
          }
        }
      }
    },
  };
}

export default {
  globs: ['src/**/*.js'],
  exclude: ['src/**/*.test.js'],
  outdir: '.',
  litelement: false,
  plugins: [stripPrivateMembers(), addCssParts()],
};
```

```js
test('gl-wc element upgrades and exposes its API', async ({ page }) => {
  await page.goto('/test/test-page.html');
  const isUpgraded = await page.evaluate(() => {
    const el = document.getElementById('wc');
    return el instanceof HTMLElement && typeof el.snapshot === 'function' && !!el.shadowRoot;
  });
  expect(isUpgraded).toBe(true);
});

test('a preset loads and the canvas renders non-empty pixels', async ({ page }) => {
  await page.goto('/test/test-page.html');
  await page.evaluate(() => document.getElementById('wc').ready);
  // Give the RAF loop a couple of frames.
  await page.waitForTimeout(300);
  const hasInk = await page.evaluate(() => {
    const canvas = document.getElementById('wc').shadowRoot.querySelector('canvas');
    const gl = canvas.getContext('webgl2') || canvas.getContext('webgl');
    const px = new Uint8Array(4);
    gl.readPixels(
      Math.floor(canvas.width / 2), Math.floor(canvas.height / 2),
      1, 1, gl.RGBA, gl.UNSIGNED_BYTE, px
    );
    return px[0] + px[1] + px[2] + px[3] > 0;
  });
  expect(hasInk).toBe(true);
});

test('snapshot() resolves to a PNG blob', async ({ page }) => {
  await page.goto('/test/test-page.html');
  await page.evaluate(() => document.getElementById('wc').ready);
  const type = await page.evaluate(async () => {
    const blob = await document.getElementById('wc').snapshot();
    return blob && blob.type;
  });
  expect(type).toBe('image/png');
});
```
